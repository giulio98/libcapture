169void Pipeline::feed(av::PacketUPtr packet, av::MediaType packet_type) {
170 if (!packet) throwRuntimeError("received packet is null");
171 if (!av::validMediaType(packet_type)) throwRuntimeError("failed to take packet (media type is invalid)");

173

178 if (!packets_[packet_type]) { // if previous packet has been fully processed
179 packets_[packet_type] = std::move(packet);

181 }
182 } else {
