71 x1 = GetSystemMetrics(SM_XVIRTUALSCREEN);

88 rofl = resolution_height;

297void Capturer::capture(Demuxer &demuxer) {
298 bool after_pause;
299 int64_t last_pts = 0;
300 int64_t pts_offset = 0;
301 bool adjust_pts_offset = false;
302 std::chrono::milliseconds sleep_interval(1);
303
304#if THROW_TEST_EXCEPTION
305 int counter = 0;
306#endif
307
308 while (true) {
309 {
310 std::unique_lock<std::mutex> ul{m_};
311 after_pause = paused_;
312#ifndef MACOS
313 if (paused_) demuxer.closeInput();
314#endif
315 cv_.wait(ul, [this]() { return (!paused_ || stopped_); });
316 if (stopped_) break;
317 }
318
319 if (after_pause) {
320 adjust_pts_offset = true;
321#ifdef MACOS
322 demuxer.flush();
323#else
324 demuxer.openInput();
325#endif
326 }
327
328 auto [packet, packet_type] = demuxer.readPacket();
329 if (!packet) {
330 std::this_thread::sleep_for(sleep_interval);
331 continue;
332 }
333 if (!av::validMediaType(packet_type)) throw std::runtime_error("Invalid packet type received from demuxer");
334
335 if (adjust_pts_offset) pts_offset += (packet->pts - last_pts);
336 last_pts = packet->pts;
337 if (adjust_pts_offset) {
338 adjust_pts_offset = false;
339 } else {
340 packet->pts -= pts_offset;
341 pipeline_->feed(std::move(packet), packet_type);
342 }
343
344#if THROW_TEST_EXCEPTION
345 if (++counter == 300) throw std::runtime_error("UGLY ERROR");
346#endif
347 }
348}
