53inline DictionaryUPtr map2dict(const std::map<std::string, std::string> &map) {
54 AVDictionary *dict = nullptr;
55 for (const auto &[key, val] : map) {
56 if (av_dict_set(&dict, key.c_str(), val.c_str(), 0) < 0) {
57 if (dict) av_dict_free(&dict);
58 throw std::runtime_error("Cannot set " + key + "in dictionary");
59 }
60 }
61 return DictionaryUPtr(dict);
62}

64inline std::map<std::string, std::string> dict2map(const AVDictionary *dict) {
65 std::map<std::string, std::string> map;
66 const AVDictionaryEntry *entry = nullptr;
67 do {
68 entry = av_dict_get(dict, "", entry, AV_DICT_IGNORE_SUFFIX);
69 if (entry) map.insert({entry->key, entry->value});
70 } while (entry);
71 return map;
