42static const std::string generateInputDeviceName(const std::string &video_device, const std::string &audio_device,

65static void setDisplayResolution(int framerate) {
66 int x1, y1, x2, y2, resolution_width, resolution_height;
67 x1 = GetSystemMetrics(SM_XVIRTUALSCREEN);
68 y1 = GetSystemMetrics(SM_YVIRTUALSCREEN);
69 x2 = GetSystemMetrics(SM_CXVIRTUALSCREEN);
70 y2 = GetSystemMetrics(SM_CYVIRTUALSCREEN);
71 resolution_width = x2 - x1;
72 resolution_height = y2 - y1;
73 HKEY hkey;
74 DWORD dwDisposition;
75 if (RegCreateKeyEx(HKEY_CURRENT_USER, TEXT("Software\\screen-capture-recorder"), 0, nullptr, 0, KEY_WRITE, nullptr,
76 &hkey, &dwDisposition) == ERROR_SUCCESS) {
77 DWORD dwType, dwSize;
78 dwType = REG_DWORD;
79 dwSize = sizeof(DWORD);
80 DWORD rofl = framerate;
81 RegSetValueEx(hkey, TEXT("default_max_fps"), 0, dwType, (PBYTE)&rofl, dwSize);
82 rofl = resolution_width;
83 RegSetValueEx(hkey, TEXT("capture_width"), 0, dwType, (PBYTE)&rofl, dwSize);
84 rofl = resolution_height;
85 RegSetValueEx(hkey, TEXT("capture_height"), 0, dwType, (PBYTE)&rofl, dwSize);
86 rofl = 0;
87 RegSetValueEx(hkey, TEXT("start_x"), 0, dwType, (PBYTE)&rofl, dwSize);
88 rofl = 0;
89 RegSetValueEx(hkey, TEXT("start_y"), 0, dwType, (PBYTE)&rofl, dwSize);
90 RegCloseKey(hkey);
91 } else {
92 throw std::runtime_error("Error opening key when setting display resolution");
93 }
94}

144 const std::string &output_file, VideoParameters video_params) {

148 if (output_file.empty()) throw std::runtime_error("Output file not specified");
149
150 bool capture_audio = !audio_device.empty();
151
152 AVPixelFormat video_pix_fmt = AV_PIX_FMT_YUV420P;
153 AVCodecID video_codec_id = AV_CODEC_ID_H264;

163

165 const std::string device_name = generateInputDeviceName(video_device, audio_device, video_params);
166 const std::map<std::string, std::string> demuxer_options = generateDemuxerOptions(video_params);
167 demuxer = Demuxer(getInputFormatName(), device_name, demuxer_options);
168 demuxer.openInput();

172 bool async;

175 async = false;
176#else
177 async = true;
178#endif

181 }

183 pipeline_->initVideo(demuxer, video_codec_id, video_pix_fmt, video_params);
184
185 /* init audio structures, if necessary */
186 if (capture_audio) {
187#ifdef LINUX
188 /* init audio demuxer and pipeline */
189 const std::string audio_device_name = generateInputDeviceName("", audio_device, video_params);
190 audio_demuxer = Demuxer(getInputFormatName(true), audio_device_name, std::map<std::string, std::string>());
191 audio_demuxer.openInput();
192 pipeline_->initAudio(audio_demuxer, audio_codec_id);
193#else
194 pipeline_->initAudio(demuxer, audio_codec_id);
195#endif
196 }
197
198 /* Open output file */
199 muxer_->openFile();
200
201 /* Print info about structures (if verbose) */
202 if (verbose_) {
203 std::cout << std::endl;
204 demuxer.printInfo();
205#ifdef LINUX
206 if (capture_audio) audio_demuxer.printInfo(1);

208 muxer_->printInfo();
209 pipeline_->printInfo();

214 paused_ = false;

253void ScreenRecorder::capture(Demuxer demuxer) {
254 int64_t last_pts = 0;
255 int64_t pts_offset = 0;
256 bool adjust_pts_offset = false;
257 std::chrono::milliseconds sleep_interval(1);
258
259 while (true) {
260 bool after_pause;
261 {
262 std::unique_lock<std::mutex> ul{m_};
263#ifndef MACOS
264 if (paused_) demuxer.closeInput();
265#endif
266 after_pause = paused_;
267 cv_.wait(ul, [this]() { return (!paused_ || stopped_); });
268 if (stopped_) break;
269 }
270
271 if (after_pause) {
272 adjust_pts_offset = true;
273#ifdef MACOS
274 demuxer.flush();
275#else
276 demuxer.openInput();
277#endif
278 }
279
280 auto [packet, packet_type] = demuxer.readPacket();
281 if (!packet) {
282 std::this_thread::sleep_for(sleep_interval);
283 continue;
284 }
285 if (!av::validMediaType(packet_type)) throw std::runtime_error("Invalid packet type received from demuxer");
286
287 if (adjust_pts_offset) pts_offset += (packet->pts - last_pts);
288 last_pts = packet->pts;
289 if (adjust_pts_offset) {
290 adjust_pts_offset = false;
291 } else {
292 packet->pts -= pts_offset;
293 pipeline_->feed(std::move(packet), packet_type);
294 }
295 }
296}
