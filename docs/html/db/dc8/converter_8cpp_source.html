14static std::pair<std::string, std::string> getAudioFilterSpec(const AVCodecContext *dec_ctx,
15 const AVCodecContext *enc_ctx, AVRational in_time_base) {
16 if (!dec_ctx) throwError("dec_ctx is NULL");
17 if (!enc_ctx) throwError("enc_ctx is NULL");
18
19 std::stringstream src_args_ss;
20 src_args_ss << "time_base=" << in_time_base.num << "/" << in_time_base.den;
21 src_args_ss << ":sample_rate=" << dec_ctx->sample_rate;
22 src_args_ss << ":sample_fmt=" << dec_ctx->sample_fmt;
23 if (dec_ctx->channel_layout) { /* if a specific channel layout is known to the decoder, use it */
24 src_args_ss << ":channel_layout=" << dec_ctx->channel_layout;
25 } else { /* otherwise, just set the number of channels */
26 src_args_ss << ":channels=" << dec_ctx->channels;
27 }
28
29 std::stringstream filter_spec_ss;
30 /* set PTS */
31 filter_spec_ss << "asetpts=PTS-STARTPTS";
32 /*
33 * convert sample_rate, sample_fmt and channel layout
34 * the async=1 option tells the filter to add extra silence / stretch / squeeze the audio
35 * to ensure that the timestamps are correct
36 */
37 filter_spec_ss << ",aresample=" << enc_ctx->sample_rate << ":async=1"
38 << ":out_sample_fmt=" << enc_ctx->sample_fmt << ":out_channel_layout=" << enc_ctx->channel_layout;
39 /* ensure correct number of samples for output frames (even with injected silence) */
40 filter_spec_ss << ",asetnsamples=n=" << enc_ctx->frame_size;
41
42 /* format conversion (OLD, now use aresample instead) */
43 // filter_spec_ss << ",aformat=sample_fmts=" << av_get_sample_fmt_name(enc_ctx->sample_fmt)
44 // << ":sample_rates=" << enc_ctx->sample_rate << ":channel_layouts=" << enc_ctx->channel_layout;
45
46 return std::make_pair(src_args_ss.str(), filter_spec_ss.str());
47}

72Converter::Converter(const AVCodecContext *dec_ctx, const AVCodecContext *enc_ctx, AVRational in_time_base,
73 int offset_x, int offset_y) {
74 if (!dec_ctx) throwError("dec_ctx is NULL");
75 if (!enc_ctx) throwError("enc_ctx is NULL");
76
77 std::string src_filter_name;
78 std::string sink_filter_name;
79 std::string src_args;
80 std::string filter_spec;
81
82 if (dec_ctx->codec_type != enc_ctx->codec_type) {
83 throwError("type mismatch between received decoder and encoder");
84 } else if (dec_ctx->codec_type == AVMEDIA_TYPE_VIDEO) {
85 src_filter_name = "buffer";
86 sink_filter_name = "buffersink";
87 std::tie(src_args, filter_spec) = getVideoFilterSpec(dec_ctx, enc_ctx, in_time_base, offset_x, offset_y);
88 } else if (dec_ctx->codec_type == AVMEDIA_TYPE_AUDIO) {
89 if (offset_x || offset_y) throwError("video offset specified specified for audio converter constructor");
90 src_filter_name = "abuffer";
91 sink_filter_name = "abuffersink";
92 std::tie(src_args, filter_spec) = getAudioFilterSpec(dec_ctx, enc_ctx, in_time_base);
93 } else {
94 throwError("unknown media type received in constructor");
95 }
96
97 filter_graph_ = av::FilterGraphUPtr(avfilter_graph_alloc());
98 if (!filter_graph_) throwError("failed to allocate filter graph");
99
100 { /* buffer src set-up*/
101 const AVFilter *filter = avfilter_get_by_name(src_filter_name.c_str());
102 if (!filter) throwError("failed to find src filter definition");
103 if (avfilter_graph_create_filter(&buffersrc_ctx_, filter, "in", src_args.c_str(), nullptr,
104 filter_graph_.get()) < 0)
105 throwError("failed to create src filter");
106 }
107
108 { /* buffer sink set-up*/
109 const AVFilter *filter = avfilter_get_by_name(sink_filter_name.c_str());
110 if (!filter) throwError("failed to find sink filter definition");
111 if (avfilter_graph_create_filter(&buffersink_ctx_, filter, "out", nullptr, nullptr, filter_graph_.get()) < 0)
112 throwError("failed to create src filter");
113 }
114
115 {
116 /* Endpoints for the filter graph. */
117 av::FilterInOutUPtr outputs(avfilter_inout_alloc());
118 if (!outputs) throwError("failed to allocate filter outputs");
119 outputs->name = av_strdup("in");
120 outputs->filter_ctx = buffersrc_ctx_;
121 outputs->pad_idx = 0;
122 outputs->next = nullptr;
123
124 av::FilterInOutUPtr inputs(avfilter_inout_alloc());
125 if (!inputs) throwError("failed to allocate filter inputs");
126 inputs->name = av_strdup("out");
127 inputs->filter_ctx = buffersink_ctx_;
128 inputs->pad_idx = 0;
129 inputs->next = nullptr;
130
131 AVFilterInOut *outputs_raw = outputs.release();
132 AVFilterInOut *inputs_raw = inputs.release();
133 int ret =
134 avfilter_graph_parse_ptr(filter_graph_.get(), filter_spec.c_str(), &inputs_raw, &outputs_raw, nullptr);
135 outputs = av::FilterInOutUPtr(outputs_raw);
136 inputs = av::FilterInOutUPtr(inputs_raw);
137 if (ret < 0) throwError("failed to parse pointers");
138 }
139
140 if (avfilter_graph_config(filter_graph_.get(), nullptr) < 0) throwError("failed to configure the filter graph");
141}
