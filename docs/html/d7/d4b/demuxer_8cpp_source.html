94std::pair<av::PacketUPtr, av::MediaType> Demuxer::readPacket() {
95 if (!fmt_ctx_) throwError("failed to read packet (input is not open)");
96
97 if (!packet_) {
98 packet_ = av::PacketUPtr(av_packet_alloc());
99 if (!packet_) throwError("failed to allocate packet");
100 }
101
102 auto packet_type = av::MediaType::None;
103
104 int ret = av_read_frame(fmt_ctx_.get(), packet_.get());
105 if (ret == AVERROR(EAGAIN)) return std::make_pair(nullptr, packet_type);
106 if (ret < 0) throwError("failed to read a packet");
107
108 for (auto type : av::validMediaTypes) {
109 if (streams_[type] && packet_->stream_index == streams_[type]->index) {
110 packet_type = type;
111 break;
112 }
113 }
114 if (packet_type == av::MediaType::None) throwError("unknown packet stream index");
115
116 return std::make_pair(std::move(packet_), packet_type);
